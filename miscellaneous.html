function calculatePointsPerSeries(matchesPlayed, matchesWon, matchesDrawn) {
const pointsPerWin = 60 / matchesPlayed;
const pointsPerDraw = pointsPerWin / 2;
const totalPoints = matchesWon * pointsPerWin + matchesDrawn * pointsPerDraw;
return totalPoints;
}

// Function to populate the table with data for all teams
function populateTable() {
const tableBody = document.querySelector("#matchResults tbody");

// Clear existing rows
tableBody.innerHTML = "";

// Loop through team data and create table rows
teamData.forEach((team) => {
    let totalMatchesPlayed = 0;
    let totalMatchesWon = 0;
    let totalMatchesLost = 0;
    let totalMatchesDrawn = 0;
    let totalPoints = 0;

    team.series.forEach((series) => {
    // Increment total matches played
    totalMatchesPlayed += series.matchesPlayed;

    // Increment total matches won, lost, and drawn
    totalMatchesWon += series.matchesWon;
    totalMatchesLost += series.matchesLost;
    totalMatchesDrawn += series.matchesDrawn;

    // Calculate points for the series
    const points = calculatePointsPerSeries(
        series.matchesPlayed,
        series.matchesWon,
        series.matchesDrawn
    );
    totalPoints += points;
    });

    // Create table row for team's total data
    const row = document.createElement("tr");
    row.innerHTML = `
        ${team.name}
        ${totalMatchesPlayed}
        ${totalMatchesWon}
        ${totalMatchesLost}
        ${totalMatchesDrawn}
        ${totalPoints.toFixed(2)}
    `;
    tableBody.appendChild(row);
});
}

// Call the function to populate the table
populateTable();
